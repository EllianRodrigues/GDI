CREATE TABLE Quarto (
    Numero_quarto INT PRIMARY KEY,
    CPF_h INT,
    Nome_tipo varchar(255),
    Status varchar(255),
    Reservado varchar(255),
    FOREIGN KEY (CPF_h) REFERENCES Hospedes(CPF_p),
    FOREIGN KEY (Nome_tipo) REFERENCES Tipos_quarto(Nome_tipo)
);
