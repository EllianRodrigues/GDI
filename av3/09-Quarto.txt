CREATE TABLE Quarto (
    numero_quarto int,
    nome_tipo VARCHAR2(20),
    status VARCHAR2(20),
    reservado VARCHAR2(5),
    PRIMARY KEY (numero_quarto),
    FOREIGN KEY (nome_tipo) REFERENCES TipoQuarto(nome_tipo)
);
