CREATE TABLE Contrata (
    CPF_h INT,
    Reserva_id INT,
    Numero_quarto INT,
    Servico_id INT,
    FOREIGN KEY (CPF_h) REFERENCES Hospedes(CPF_p),
    FOREIGN KEY (Reserva_id) REFERENCES Reserva(Reserva_id),
    FOREIGN KEY (Numero_quarto) REFERENCES Quarto(Numero_quarto),
    FOREIGN KEY (Servico_id) REFERENCES Servico_hospede(Servico_id)
);
