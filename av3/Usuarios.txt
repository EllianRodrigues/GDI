CREATE TABLE Usuarios ( 
    CPF int NOT NULL PRIMARY KEY,  
    Nome varchar(255),  
    Rua varchar(255),  
    Numero varchar(255),  
    Complemento varchar(255),  
    Bairro varchar(255),  
    Cidade varchar(255),  
    Estado varchar(255),  
    Pais varchar(255),  
    Cep int 
);
